var harvest_pick = false
var speed_mode = true
var quarterstaff_use = true

func set_harvest_pick(b)
  harvest_pick = b

func set_speed_mode(b)
  speed_mode = b

func set_quarterstaff_use(b)
  quarterstaff_use = b

func action()
  ?harvest_pick & harvest.distance <= 9
      harvest_walk()
  :?pickup & pickup.distance <= 9
      pick_walk()
  :?speed_mode
    speed_walk()
  :
    ?hp = maxhp
      speed_walk()
    :
      heal_walk()

func harvest_walk()
  ?harvest = Boulder
    equip shovel
  :
    // todo: hatchet のactive を振りたい
    equipL triskelion stone
    equipR hatchet

func pick_walk()
  equipL triskelion stone
  equipR star_stone

func heal_walk()
  equip quarterstaff
  ?(time > 10 & item.CanActivate() & item.GetCooldown("quarterstaff") = 0) | buffs.string = quarterstaff & quarterstaff_use
    activate R
  :
    equipL ouroboros
    equipR com shield

func speed_walk()
  equip quarterstaff
  ?(time > 10 & item.CanActivate() & item.GetCooldown("quarterstaff") = 0) | buffs.string = quarterstaff & quarterstaff_use
    activate R
  :
    equipL triskelion stone
    equipR com shield
